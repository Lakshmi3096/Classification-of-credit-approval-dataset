%attribute value 'b'.

         count_cell_neg(2,1)=count_cell_neg(2,1)+1;
     else
         count_empty(1,1)=count_empty(1,1)+1;
     end
 end

 %of occurences of 'a' in '+' class is higher or else 'b'. The same is repeated
 % for the instance of class '-'.
